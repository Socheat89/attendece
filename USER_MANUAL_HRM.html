ប្រព័ន្ធគ្រប់គ្រងធនធានមនុស្សសរុប

ឯកសារនេះបង្ហាញពីជំហាននៃការប្រើប្រាស់ប្រព័ន្ធ HRM ចាប់ពីការចូលគណនី រហូតដល់ការគណនា និងបើកប្រាក់ខែ (End-to-End User Flow)។

១. សម្រាប់អ្នកគ្រប់គ្រង (Admin / HR)

ជំហានរៀបចំប្រព័ន្ធ គ្រប់គ្រងបុគ្គលិក វត្តមាន និងបើកប្រាក់ខែ

1

ការចូលប្រើប្រាស់ (Login)

  • ចូលទៅកាន់ URL របស់ប្រព័ន្ធ
  • បញ្ចូល Email និង Password របស់ Admin
  • Dashboard នឹងបង្ហាញទិន្នន័យសង្ខេប (ចំនួនបុគ្គលិក វត្តមានថ្ងៃនេះ)
2

រៀបចំក្រុមហ៊ុន (Company Setup)

  • បង្កើត សាខា (Branches) ទីតាំងការងារ
  • បង្កើត ផ្នែក (Departments)
  • បង្កើត តួនាទី (Designations)
  • កំណត់ ម៉ោងការងារ (Schedules)
3

គ្រប់គ្រងបុគ្គលិក (Employee)

  • ចូលទៅ Employees > Add New
  • បំពេញព័ត៌មាន (ឈ្មោះ, អ៊ីមែល, លេខកាត)
  • ភ្ជាប់ Employee ទៅផ្នែក និងម៉ោងការងារ
  • បង្កើតគណនី (Login Info) ឱ្យគាត់
4

គ្រប់គ្រងវត្តមាន (Attendance)

  • ទាញយក/Print QR Code ស្កេនវត្តមាន
  • ពិនិត្យបញ្ជីវត្តមាន (Late, Early, Absent)
  • បញ្ចូលវត្តមានដោយដៃ បើមានការខុសឆ្គងពីប្រព័ន្ធ
5

ច្បាប់ និងថែមម៉ោង (Leaves & OT)

  • ចូលទៅ Leave Options > Requests
  • ពិនិត្យការសុំច្បាប់ > ចុច Approve / Reject
  • ពិនិត្យការថែមម៉ោង ដើម្បីទូទាត់ប្រាក់នៅចុងខែ
6

គណនាប្រាក់ខែ (Payroll)

  • ចូល Payroll > រើសខែ/ឆ្នាំ > Generate
  • ប្រព័ន្ធនឹងគណនា Base Salary + OT - Leaves/Late
  • ពិនិត្យរួច ចុច Approve & Lock ដើម្បីឱ្យបុគ្គលិកមើលឃើញ

២. សម្រាប់បុគ្គលិក (Employee)

ការប្រើប្រាស់គណនីផ្ទាល់ខ្លួន ស្កេនកាត សុំច្បាប់ និងមើលប្រាក់ខែ

ការចូលប្រើ និងស្កេនម៉ោង

ក. ការចូលប្រើ (Login)

ទទួលគណនីពី HR ហើយ Login ចូល Dashboard របស់បុគ្គលិកដោយផ្ទាល់។

ខ. ការស្កេន ឬវាយកាតចូលរៀន (Scan In/Out)

  • • ចូលទៅម៉ឺនុយ Scan.
  • • ប្រើកាមេរ៉ាទូរស័ព្ទស្កេនមេ QR Code របស់ក្រុមហ៊ុន។
  • • ប្រព័ន្ធនឹងឆ្លើយតបជោគជ័យ (Check In ពេលព្រឹក, Check Out ពេលល្ងាច)។

គ. ពិនិត្យវត្តមាន (My Attendance)

ចូល Report > My Attendance ដើម្បីមើលប្រវត្តិវត្តមានប្រចាំថ្ងៃ និងដឹងពីថ្ងៃដែលយឺត។

សំណើសុំ និងប្រាក់ខែ

ឃ. ស្នើសុំច្បាប់ឈប់ (Request Leave)

  • • ចូលម៉ឺនុយ Leave > Apply.
  • • ជ្រើសរើសប្រភេទ (ឈឺ, ធុរៈ), កាលបរិច្ឆេទ និងមូលហេតុ រួចចុច Submit។
  • • រង់ចាំការយល់ព្រមពី HR តាមរយៈ Status ក្នុងប្រព័ន្ធ។

ង. ស្នើសុំថែមម៉ោង (Request OT)

ចូលម៉ឺនុយ Overtime > បញ្ចូលកាលបរិច្ឆេទ, ចំនួនម៉ោង និងមូលហេតុដើម្បីទូទាត់ប្រាក់ខែ។

ច. មើលប្រាក់ខែ (Payslip)

  • • ចុងខែ ចូលទៅកាន់ម៉ឺនុយ Salary / Payslip.
  • • ចុច View ឬ Download PDF ដើម្បីមើលកន្ទុយប្រាក់ខែគិតរួចរាល់។
